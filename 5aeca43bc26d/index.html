P1064 йҮ‘жҳҺзҡ„йў„з®—ж–№жЎҲ

иҝҷйўҳпјҢеҲҡзңӢиө·жқҘеҫҲйҡҫпјҢеӣ дёәиҰҒеӨ„зҗҶдё»д»¶дёҺйҷ„д»¶зҡ„е…ізі»гҖӮе…¶е®һпјҢиӢҘиҝҗз”ЁеҲҶз»„зҡ„жҖқжғіпјҢиҝҷдёӘйўҳеҫҲе®№жҳ“гҖӮ

еҜ№дәҺиҝҷдёӘйўҳпјҢ$c$е°ұжҳҜзү©е“Ғзҡ„д»·ж јдёҺйҮҚиҰҒеәҰзҡ„д№ҳз§ҜпјҢ$w$е°ұжҳҜзү©е“Ғзҡ„д»·ж јгҖӮ

ж—ўз„¶жҜҸдёӘдё»д»¶еҸҜд»Ҙжңү$0$дёӘгҖҒ$1$дёӘжҲ–$2$дёӘйҷ„д»¶пјҢеҲҷиҙӯд№°е®ғд»¬зҡ„жғ…еҶөеҲҶдёә$4$з§Қпјҡ

  • д№°дёҖдёӘдё»д»¶гҖӮ
  • д№°дёҖдёӘдё»д»¶дёҺдёҖдёӘйҷ„д»¶гҖӮ
  • д№°дёҖдёӘдё»д»¶дёҺеҸҰдёҖдёӘйҷ„д»¶гҖӮ
  • д№°дёҖдёӘдё»д»¶дёҺдёӨдёӘйҷ„д»¶гҖӮ

е№¶дё”пјҢеңЁиҝҷдәӣиҙӯд№°ж–№жЎҲдёӯпјҢеҸӘиғҪйҖүдёҖдёӘгҖӮ

иҝҷж ·пјҢжҲ‘д»¬еҸҜд»ҘеҖҹйүҙе•ҶеңәйҮҢзҡ„жү“еҢ…й”Җе”®пјҢжҠҠдёҠйқўжҜҸз§Қжғ…еҶөжүҖж¶үеҸҠеҲ°зҡ„зү©е“ҒеҲҶеҲ«жҚҶжҲҗдёҖжҚҶпјҢеҪ“дҪң$4$дёӘзү©е“ҒгҖӮиҝҷ$4$дёӘзү©е“ҒдёӯпјҢиҰҒд№ҲеҸӘйҖүдёҖдёӘпјҢиҰҒд№ҲдёҚйҖүпјҲеҲҶз»„пјҒпјҒпјҒпјүгҖӮ

и®ҫжңү$q$дёӘдё»д»¶пјҢеҲҷжңү$q$з»„пјҒ

CODEпјҡ

#include<bits/stdc++.h> 
using namespace std;
int m,n,w[60][10],c[60][10],tol[60],f[32000+5]; 
int main()
{
    scanf("%d%d",&m,&n);
    for(int i=1;i<=n;i++)
    {
        int a,b,z;
        scanf("%d%d%d",&a,&b,&z);
        if(z==0)//дё»д»¶
            w[i][1]=a,c[i][1]=a*b,tol[i]=1;
        else
            if(tol[z]==1)//з¬¬дёҖдёӘйҷ„д»¶
                w[z][2]=a+w[z][1],c[z][2]=a*b+c[z][1],tol[z]=2;
            else//з¬¬дәҢдёӘйҷ„д»¶
                w[z][3]=a+w[z][1],c[z][3]=a*b+c[z][1],w[z][4]=a+w[z][2],c[z][4]=a*b+c[z][2],tol[z]=4;
    }
    for(int i=1;i<=n;i++)//еҲҶз»„иғҢеҢ…дёҚз”ЁеӨҡиҜҙ
        for(int j=m;j>=0;j--)
            for(int k=0;k<=tol[i];k++)
                if(j>=w[i][k])
                    f[j]=max(f[j],f[j-w[i][k]]+c[i][k]);

    printf("%d",f[m]);
    return 0;
}
