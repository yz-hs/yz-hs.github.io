CF762A k-th divisor

题意：找到 nnn 的第 kkk 小因子。不存在输出 −1-1−1。

数据范围：1≤n≤10151\leq n\leq 10^{15}1≤n≤1015 ，1≤k≤1091\leq k\leq 10^91≤k≤109。

题解与代码

题解：注意到因子都是成对出现的。因此枚举 [1,n][1,\sqrt{n}][1,n​] 暴力判断即可。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	long long n,k,cnt=0,exist=0; 
	void yzmain()
	{
		scanf("%lld%lld",&n,&k);
		for(long long i=1;i*i<=n;i++)//暴力判断
			if(n%i==0)
			{
				cnt++;if(i*i==n) exist=1;//exist 判断是否存在两个相同数相乘为n
				if(cnt==k)
				{
					printf("%lld\n",i);
					return;
				}
			}
		cnt*=2;cnt-=exist;//算出因子数
		if(cnt<k)//不存在输出 -1
		{
			printf("-1\n");
			return;
		}
		for(long long i=1;i*i<=n;i++)//接着暴力判断
			if(n%(n/i)==0)
			{
				if(cnt==k)
				{
					printf("%lld\n",n/i);
					return;
				}
				cnt--;
			}
		}
		return;
	}
}
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	_yz::yzmain();
	return 0;
}

CF776B Sherlock and his girlfriend

题意：有 nnn 个点，编号 [2,n+1][2,n+1][2,n+1]，给这些点染色，要求若 aaa 是 bbb 的质因子，则 aaa 和 bbb 的颜色不同，求一种颜色数最少的方案。

数据范围：n≤105n\leq 10^5n≤105。

题解与代码

题解：注意到质因子关系是二分图，一边是质数，一边是合数。把质数都染成 111，合数都染成 222 即可。因此当 n≤2n \leq 2n≤2 时，颜色数最少为 111，否则为 222。惊不惊喜，意不意外？

代码：

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	int n;
	int check(int x)
	{
		for(int i=2;i*i<=x;i++)
			if(x%i==0) return 2;
		return 1;
	}
	void yzmain()
	{
		scanf("%d",&n);
		if(n==1)
		{
			printf("1\n1");
			return;
		}
		if(n==2)
		{
			printf("1\n1 1");
			return; 
		}
		printf("2\n");
		for(int i=2;i<=n+1;i++)
			printf("%d ",check(i));
		return;
	}
}
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	_yz::yzmain();
	return 0;
}

CF664A Complicated GCD

题意：给定整数 aaa，bbb，求 gcd⁡(a,a+1,a+2,…,b)\gcd(a, a+1, a+2, …, b)gcd(a,a+1,a+2,…,b)。

数据范围：1≤a≤b≤101001\leq a\leq b\leq10^{100}1≤a≤b≤10100。

题解与代码

题解：

设 (a,b)(a,b)(a,b) 表示 a,ba,ba,b 的最大公约数。

注意到，(a+1,a)=(a,1)=(1,0)=1(a+1,a)=(a,1)=(1,0)=1(a+1,a)=(a,1)=(1,0)=1。故当 a<ba < ba<b 时，输出 111，否则输出 aaa。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	string a,b; 
	void yzmain()
	{
		cin>>a>>b;
		if(a!=b) printf("1\n");
		else cout<<a;
		return;
	}
}

{

CF757B Bash’s Big Day

题意：给定 nnn 个正整数 aia_iai​。求一个子集 SSS，满足 gcd⁡(S1,...,Sk)>1\gcd(S_1,...,S_k)>1gcd(S1​,...,Sk​)>1，同时 ∣S∣|S|∣S∣ 尽可能大。

数据范围：1≤n,ai≤1051\leq n, a_i\leq 10^51≤n,ai​≤105。

题解与代码

题解：gcd⁡>1\gcd>1gcd>1，说明存在一个正整数 d>1d>1d>1，满足 ddd 整除 SSS 内的所有元素。枚举 d∈[2,max⁡{ai}]d \in [2,\max\{a_i\}]d∈[2,max{ai​}] ，以桶的方式统计答案。设 V=max⁡{ai}V=\max\{a_i\}V=max{ai​} ，时间复杂度为 O(Vln⁡V)O(V \ln V)O(VlnV)。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	int n,a[100000+5],ans=1,mx=0,tmp;
	int check(int k)
	{
		int ret=0;
		for(int i=k;i<=mx;i+=k)
			ret+=a[i];
		return ret;
	}
	void yzmain()
	{
		scanf("%d",&n);
		for(int i=1;i<=n;i++) scanf("%d",&tmp),mx=max(mx,tmp),a[tmp]++;
		for(int i=2;i<=mx;i++) ans=max(ans,check(i));
		printf("%d\n",ans);
		return;
	}
}
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	_yz::yzmain();
	return 0;
}

CF632D Longest Subsequence

题意：给出包含 nnn 个数的数列 AAA，要求选出尽可能多的数，满足它们的最小公倍数不大于 mmm 。允许不选，此时最小公倍数为 111。输出这个最小公倍数，选出数的个数，以及这些数在数列中的位置。

数据范围：n,m≤106n,m\le{10}^{6}n,m≤106，ai≤109a_i\le{10}^{9}ai​≤109。

题解与代码

题解：

首先，注意到 m≤106m\le10^6m≤106，所以若 ai>ma_i>mai​>m，则 aia_iai​ 可直接舍弃。好，我们成功地把数据范围由 10910^9109 优化到 10610^6106 了。

之后，如果数 kkk 的约数集合为 SSS，则集合 SSS 的最小公倍数一定小于等于 kkk。好，我们成功地把问题转化为求约数个数了。

我们可以枚举 x∈[1,m]x\in[1,m]x∈[1,m]，求数列 AAA 中有多少数可整除 xxx，求个最大值。不过这样时间复杂度为 O(nm)O(nm)O(nm)，超时。我们可以把这个过程转换一下，枚举 x∈Ax\in Ax∈A，看 xxx 能为 [1,m][1,m][1,m] 中的哪些数做贡献。很明显，xxx 可为所有为 xxx 的倍数的数做贡献（当然要小于 mmm）。好，我们成功地降低时间复杂度为 O(nlog⁡m)O(n\log m)O(nlogm) 了。

此算法可通过本题。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	const int N=1000000+10;
	int a[N],num[N],n,m,ans[N],v[N],lcm=1,cnt=0;
	void yzmain()
	{
		scanf("%d%d",&n,&m);
		for(int i=1;i<=n;i++) 
		{
			scanf("%d",a+i);
			if(a[i]<=m) num[a[i]]++;//忽略大于m的数，使用桶统计数的个数
		}
		for(int i=1;i<=n;i++)
		{
			if(a[i]>m || v[a[i]]) continue;//数大于m || 重复数
			for(int j=1;j*a[i]<=m;j++)
				ans[a[i]*j]+=num[a[i]];//统计贡献
			v[a[i]]=1;//去重
		}
		for(int i=1;i<=m;i++)
			if(ans[i]>cnt) lcm=i,cnt=ans[i];//保证是**最小**公倍数
		printf("%d %d\n",lcm,cnt);
		for(int i=1;i<=n;i++)
			if(lcm%a[i]==0) printf("%d ",i);
		return;
	}
}
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	_yz::yzmain();
	return 0;
}

P1445 [Violet]樱花

题意：给定 nnn，求方程 1x+1y=1n!\dfrac{1}{x} + \dfrac{1}{y} = \dfrac{1}{n!}x1​+y1​=n!1​ 的正整数解的组数，答案对 109+710^9+7109+7 取模。1≤n≤1061\le n \le10^61≤n≤106。

点击查看题解

题解：看着这个式子，什么玩意！不过可以对式子进行变形：

1x+1y=1n!xn!+yn!=xy0=−xn!−yn!+xy(n!)2=(n!)2−xn!−yn!+xy(n!)2=(x−n!)(y−n!)\begin{aligned} \dfrac{1}{x} + \dfrac{1}{y} &= \dfrac{1}{n!}\\\\ xn! + yn! &= xy\\\\ 0 &= - xn! - yn! + xy\\\\ (n!)^2 &= (n!)^2 -xn! -yn! +xy\\\\ (n!)^2 &= (x-n!)(y-n!)\\\\ \end{aligned}x1​+y1​xn!+yn!0(n!)2(n!)2​=n!1​=xy=−xn!−yn!+xy=(n!)2−xn!−yn!+xy=(x−n!)(y−n!)​

发现，每一组 xxx 和 yyy 都唯一对应着 (n!)2(n!)^2(n!)2 的因子。

所以，此题转化为，求 (n!)2(n!)^2(n!)2 的因子个数。

算术基本定理：N=∏i=1kpieiN = \prod\limits_{i=1}^k p_i^{e_i}N=i=1∏k​piei​​。 pip_ipi​ 为质数。这样的分解是唯一的。

因数个数：M=∏i=1k(ei+1)M = \prod\limits_{i=1}^k (e_i+1)M=i=1∏k​(ei​+1)。可以推出 M2=∏i=1k(2ei+1)M^2 = \prod\limits_{i=1}^k (2e_i+1)M2=i=1∏k​(2ei​+1)。排列组合和乘法原理可证。

附加题：求 ∑xi+yi\sum x_i+y_i∑xi​+yi​。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace name
{
	typedef long long ll;
	const ll mod=1000000007,N=1000000+10;
	ll n,prime[N],p[N],e[N],ans=1,cnt=0;
	void getPrime(ll n)
	{
		for(ll i=2;i<=n;i++)
		{
			if(!p[i]) prime[++cnt]=i,p[i]=i;
			for(ll j=1;j<=cnt;j++)
			{
				if(prime[j]>p[i]||prime[j]>n/i) break;
				// 保证 prime_j 是最小质因数 || 不要超过 n 
				p[i*prime[j]]=prime[j];
			}
		}
		return;
	}
	void main()
	{
		scanf("%lld",&n);
		getPrime(n);
		for(ll i=1;i<=n;i++)
		{
			for(ll j=i;j>=2;j/=p[j])
				e[p[j]]++;
		}
		for(ll i=2;i<=n;i++)
			ans=(ans*(e[i]*2%mod+1))%mod;
		printf("%lld\n",ans);
		return;
	}
}
int main()
{
	name::main();
	return 0;
}

P1072 [NOIP2009 提高组] Hankson 的趣味题

题意：已知正整数 a0a_0a0​，a1a_1a1​，b0b_0b0​，b1b_1b1​，设某未知正整数 xxx 满足：

  1. xxx 和 a0a_0a0​ 的最大公约数是 a1a_1a1​；

  2. xxx 和 b0b_0b0​ 的最小公倍数是 b1b_1b1​。

求解满足条件的 xxx 的个数。保证有 1≤a0,a1,b0,b1≤2×1091 \leq a_0,a_1,b_0,b_1 \leq 2 \times 10^91≤a0​,a1​,b0​,b1​≤2×109 ， n≤2000n\le2000n≤2000 且 a1∣a0a_1|a_0a1​∣a0​，b0∣b1b_0|b_1b0​∣b1​。

点击查看题解

题解：

  1. gcd⁡(a,b)=c⇒gcd⁡(a/c,b/c)=1\gcd(a,b) = c \Rightarrow \gcd(a/c,b/c) = 1gcd(a,b)=c⇒gcd(a/c,b/c)=1 （显然 / 反证）。
  2. lcm⁡(a,b)=c⇒gcd⁡(a,b)=a×b/c⇒gcd⁡(a/(a×b/c),b/(a×b/c))=1⇒gcd⁡(c/b,c/a)=1\operatorname{lcm}(a,b) = c \Rightarrow \gcd(a,b) = a\times b/c \Rightarrow \gcd(a/(a\times b/c),b/(a\times b/c)) = 1 \Rightarrow \gcd(c/b,c/a) = 1lcm(a,b)=c⇒gcd(a,b)=a×b/c⇒gcd(a/(a×b/c),b/(a×b/c))=1⇒gcd(c/b,c/a)=1。

现在应用这两个结论。

  1. gcd⁡(x,a0)=a1⇒gcd⁡(x/a1,a0/a1)=1\gcd(x,a_0) = a_1\Rightarrow \gcd(x/a_1,a_0/a_1) = 1gcd(x,a0​)=a1​⇒gcd(x/a1​,a0​/a1​)=1。
  2. lcm⁡(x,b0)=b1⇒gcd⁡(b1/x,b1/b0)=1\operatorname{lcm}(x,b_0) = b_1 \Rightarrow \gcd(b_1/x,b_1/b_0) = 1lcm(x,b0​)=b1​⇒gcd(b1​/x,b1​/b0​)=1。

输入数据保证 a0a_0a0​ 能被 a1a_1a1​ 整除，b1b_1b1​ 能被 b0b_0b0​ 整除。故枚举 xxx（b1b_1b1​ 因子） 即可。

代码：

#include<bits/stdc++.h>
using namespace std;
namespace name
{
	typedef long long ll;
	ll a0,a1,b0,b1,ans;
	ll gcd(ll a,ll b){return b?gcd(b,a%b):a;}
	ll check(ll x)
	{
		if(x%a1!=0) return 0;
		if(gcd(x/a1,a0/a1)!=1) return 0;
		if(gcd(b1/x,b1/b0)!=1) return 0;
		return 1;
	}
	void main()
	{
		ll T;
		scanf("%lld",&T);
		while(T--)
		{
			scanf("%lld%lld%lld%lld",&a0,&a1,&b0,&b1);
			ans=0;
			for(ll i=1;i*i<=b1;i++)
			{
				if(b1%i) continue;
				if(check(i)) ans++;
				ll j=b1/i;
				if(i==j) break;
				if(check(j)) ans++;
				
			}
			printf("%lld\n",ans);
		} 
		return;
	}
}
int main()
{
	name::main();
	return 0;
}

P3197 [HNOI2008]越狱

题意

监狱有 nnn 个房间，每个房间关押一个犯人，有 mmm 种宗教，每个犯人会信仰其中一种。如果相邻房间的犯人的宗教相同，就可能发生越狱，求有多少种状态可能发生越狱。

答案对 100003100003100003 取模。

对于 100%100\%100% 的数据，保证 1≤m≤1081 \le m \le 10^81≤m≤108，1≤n≤10121 \le n \le 10^{12}1≤n≤1012。

点击查看题解

题解

发生越狱数 === 总数 −-− 不发生越狱数。

总数：每个房间有 mmm 种宗教可选，故为 mnm^nmn。

不发生越狱数：第一个房间信什么都行，第二个不能信第一个信的那个，第三个不能信第二个信的那个……故为 m×(m−1)n−1m\times(m-1)^{n-1}m×(m−1)n−1。

所以答案为：mn−m×(m−1)n−1m^n-m\times(m-1)^{n-1}mn−m×(m−1)n−1。

代码

#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
	typedef long long ll;
	const ll mod=100003; 
	ll n,m,ans=0;
	ll pow(ll a,ll b)
	{
		ll ret=1;
		while(b)
		{
			if(b%2) ret=ret*a%mod;
			a=a*a%mod; b/=2;
		}
		return ret;
	}
	void main()
	{
		scanf("%lld%lld",&m,&n);
		ans=pow(m,n);
		ans=((ans-(m*pow(m-1,n-1)))%mod+mod)%mod;
		printf("%lld\n",ans);
		return;
	}
}
int main()
{
	_yz::main();
	return 0;
}

小白鼠保护协会的孤注一掷

2023-08-14 不想改格式。

题目背景

邪恶的 xytz 又双叒要拿小白鼠做实验了，身为小白鼠保护协会的你要尽自己最大可能去保护它们。

题目描述

xytz 有 nnn 瓶药，有一瓶过期了（过期药有毒），需要用多只小白鼠来测试哪一瓶药过期，过期药 24h 毒效发作，问：保证在时间最短的前提下，最少用多少只小白鼠。

输入格式

输入一个正整数 nnn，表示需要测试多少瓶药。

输出格式

输出一个正整数 mmm，表示在时间最短的前提下，最少用多少只小白鼠。

样例 #1

样例输入 #1

100

样例输出 #1

7

提示

对于 40%40\%40% 的数据，1≤n≤1091\le n\le 10^91≤n≤109；

对于 100%100\%100% 的数据，1≤n≤10181\le n\le 10^{18}1≤n≤1018。

题解

信息论。把 nnn 瓶药编号，如果其编号在二进制下第 iii 位为 111，就让第 iii 个小白鼠吃。这样看死亡情况即可推知是哪瓶药有毒。
