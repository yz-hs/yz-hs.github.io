线段树

作者：yz-hs。 本文章共有 3.2k 字。 本文写于 2020 年 11 月 28 日 。

本文为 yz-hs 学习线段树的笔记以及模板。

线段树， 是用来存放给定区间内对应信息的一种数据结构。可用来处理数组相应的区间查询和元素更新操作，也可以进行区间最大值，区间最小值或者区间异或值的查询。线段树可以处理很多符合结合律的操作。

简介

线段树是一种二叉搜索树，它将一个区间划分成一些单元区间，每个单元区间对应线段树中的一个叶结点。对于线段树中的每一个非叶子节点 $[a,b]$，它的左儿子表示的区间为 $[a,(a+b)/2]$，右儿子表示的区间为 $[(a+b)/2+1,b]$。因此叶节点数目为 $N$，即整个线段区间的长度。子节点不断向自己的父亲节点传递信息，而父节点存储的信息则是他的每一个子节点信息的整合。

线段树的时间复杂度可达 log 级别。而未优化的空间复杂度为 4N 级别，因此有时需要离散化让空间压缩。

模板

本模板使用 指针 的方式存储，比较好写。

#1 区间加，区间查询和

线段树要修改很多次，但是很多次修改的值并没有被用到，这几次修改就是无用的，那么我们通过加一个 lazytag 变量，记录当前需要往下传递的数值，待有用时再继续往下传递，这就是 LazyTag 思想。

点击查看代码 
const int N=100005;
struct SGT
{
    long long idata[N],n,index=0;
    struct Node
    {
        Node *lson,*rson;
        long long data,tag;
    }node[N<<3],*root=&node[0];

    void pushdown(Node *tree,int l,int r)//lazytag 下放
    {
        if(tree->tag==0) return;
        int mid=(l+r)/2;
        tree->lson->data+=tree->tag*(mid-l+1);
        tree->rson->data+=tree->tag*(r-mid);
        tree->lson->tag+=tree->tag;
        tree->rson->tag+=tree->tag;
        tree->tag=0;
        return;
    }

    void update(Node *tree,int l,int r,int x,int y,long long ndata)//区间加
    {
        if(x<=l&&y>=r)
        {
            tree->data+=ndata*(r-l+1);
            tree->tag+=ndata;
            return;
        }
        int mid=(l+r)/2;
        pushdown(tree,l,r);
        if(x<=mid) update(tree->lson,l,mid,x,y,ndata);
        if(y>mid) update(tree->rson,mid+1,r,x,y,ndata);
        tree->data=tree->lson->data+tree->rson->data;
    }

    long long query(Node *tree,int l,int r,int x,int y)//区间查询
    {
        if(x<=l&&y>=r) return tree->data;
        pushdown(tree,l,r);
        int mid=(l+r)/2;long long ansl=0,ansr=0;
        if(x<=mid) ansl=query(tree->lson,l,mid,x,y);
        if(y>mid) ansr=query(tree->rson,mid+1,r,x,y);
        return ansl+ansr;
    }

    void build(Node *tree,int l,int r)//建树
    {
        if(l==r){tree->data=idata[l];return;}
        int mid=(l+r)/2;
        tree->lson=&node[++index];//分配地址
        tree->rson=&node[++index];
        build(tree->lson,l,mid);build(tree->rson,mid+1,r);
        tree->data=tree->lson->data+tree->rson->data;
        return;
    }

    void init(int cnt,long long data[])//初始化
    {
        for(int i=1;i<=cnt;i++) idata[i]=data[i];
        n=cnt; build(root,1,n);
        return;
    }
}Tree;

#2 区间加，区间乘，区间查询和

我们发现一个 lazytag 是不能够解决问题的，那就上两个，分别表示乘法意义上的lazytag和加法意义上的lazytag。紧接着想到 pushdown 操作之后我们又发现必须在向下传递 lazytag 的时候人为地为这两个lazytag规定一个先后顺序。本代码规定乘法优先。

规定好 tree->son->data = ( tree->son->data*tree->mtag + tree->atag*(r-l+1) ) %p，这样的话假如改变 atag 的数值就只改变 atag，改变 mtag 的时候把 atag 也对应的乘一下就可以了，没有精度损失，看起来很不错。

点击查看代码 
const int N=100005;int p=19260817;//模数
struct SGT
{
    long long idata[N],n,index=0;
    struct Node
    {
        Node *lson,*rson;
        long long data,atag=0,mtag=1;
    }node[N<<3],*root=&node[0];

    void pushdown(Node *tree,int l,int r)
    {
        if((tree->atag==0&&tree->mtag==1)||tree->lson==NULL) return;
        int mid=(l+r)/2;
        tree->lson->data=(tree->lson->data*tree->mtag+tree->atag*(mid-l+1))%p;
        tree->rson->data=(tree->rson->data*tree->mtag+tree->atag*(r-mid))%p;
        tree->lson->mtag=(tree->lson->mtag*tree->mtag)%p;
        tree->rson->mtag=(tree->rson->mtag*tree->mtag)%p;
        tree->lson->atag=(tree->lson->atag*tree->mtag+tree->atag)%p;
        tree->rson->atag=(tree->rson->atag*tree->mtag+tree->atag)%p;
        tree->atag=0,tree->mtag=1;
        return;
    }

    void aupdate(Node *tree,int l,int r,int x,int y,long long ndata)//区间加
    {
        if(x<=l&&y>=r)
        {
            tree->data=(tree->data+ndata*(r-l+1))%p;
            tree->atag=(tree->atag+ndata)%p;
            return;
        }
        int mid=(l+r)/2;
        pushdown(tree,l,r);
        if(x<=mid) aupdate(tree->lson,l,mid,x,y,ndata);
        if(y>mid) aupdate(tree->rson,mid+1,r,x,y,ndata);
        tree->data=(tree->lson->data+tree->rson->data)%p;
    }

    void mupdate(Node *tree,int l,int r,int x,int y,long long ndata)//区间乘
    {
        if(x<=l&&y>=r)
        {
            if(tree->atag) pushdown(tree,l,r);
            tree->data=(tree->data*ndata)%p;
            tree->mtag=(tree->mtag*ndata)%p;
            return;
        }
        int mid=(l+r)/2;
        pushdown(tree,l,r);
        if(x<=mid) mupdate(tree->lson,l,mid,x,y,ndata);
        if(y>mid) mupdate(tree->rson,mid+1,r,x,y,ndata);
        tree->data=(tree->lson->data+tree->rson->data)%p;
    }

    long long query(Node *tree,int l,int r,int x,int y)
    {
        if(x<=l&&y>=r) return tree->data;
        pushdown(tree,l,r);
        int mid=(l+r)/2;long long ansl=0,ansr=0;
        if(x<=mid) ansl=query(tree->lson,l,mid,x,y);
        if(y>mid) ansr=query(tree->rson,mid+1,r,x,y);
        return ansl+ansr;
    }

    void build(Node *tree,int l,int r)
    {
        if(l==r){tree->data=idata[l];return;}
        int mid=(l+r)/2;
        tree->lson=&node[++index];
        tree->rson=&node[++index];
        build(tree->lson,l,mid);build(tree->rson,mid+1,r);
        tree->data=tree->lson->data+tree->rson->data;
        return;
    }

    void init(int cnt,long long data[])
    {
        for(int i=1;i<=cnt;i++) idata[i]=data[i];
        n=cnt; build(root,1,n);
        return;
    }
}Tree;

    void update(Node *tree,int l,int r,int pos,long long ndata)
    {
        if(l==pos&&l==r)
        {
            tree->data+=ndata;
            return;
        }
        int mid=(l+r)/2;
        if(pos<=mid) update(tree->lson,l,mid,pos,ndata);
        if(pos>mid) update(tree->rson,mid+1,r,pos,ndata);

           int mid=(l+r)/2;long long ansl=0,ansr=0;
           if(x<=mid) ansl=query(tree->lson,l,mid,x,y);
           if(y>mid) ansr=query(tree->rson,mid+1,r,x,y);

P1198 [JSOI2008]最大数

link。

题面

现在请求你维护一个数列，要求提供以下两种操作，共 $M$ 个操作：

  • 查询操作。

    • 语法：Q L。
    • 功能：查询当前数列中末尾 $L$ 个数中的最大的数，并输出这个数的值。
    • 限制：$L$ 不超过当前数列的长度。$(L > 0)$。

  • 插入操作。

    • 语法：A n
    • 功能：将 $n$ 加上 $t$，其中 $t$ 是最近一次查询操作的答案（如果还未执行过查询操作，则$t=0$)，并将所得结果对一个固定的常数 $D$ 取模，将所得答案插入到数列的末尾。
    • 限制：$n$ 是整数（可能为负数）并且在长整范围内。

注意：初始时数列是空的，没有一个数。

对于全部的测试点，保证 $1 \leq M \leq 2 \times 10^5$，$1 \leq D \leq 2 \times 10^9$。

题解

这很明显，是单点修改、区间查询最大值，可用线段树解决。

先建立一个叶子节点数为 $M$ 的线段树，叶子节点赋值为极小值。并定义 $len$ 表示数列长度。

插入操作即为修改第 $len+1$ 节点的值，并 $len \gets len+1$，单点修改。

查询操作即为查询区间 $[len-L+1,len]$ 的最大值，区间查询。

单点修改无需 lazytag。

点击查看代码 
#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
    const int N=200005,inf=2147483647;
    long long p,n,a[N],len=0;
    SGT Tree;//模板 #3 相信大家都会把它改成维护最大值的吧

    void yzmain()
    {
        long long initer[N+5];char t1;long long t2,t=0;
        scanf("%lld%lld",&n,&p);
        for(int i=1;i<=N;i++) initer[i]=-inf;
        Tree.init(n,initer);
        for(long long i=1;i<=n;i++)
        {
            cin>>t1>>t2;
            if(t1=='A') Tree.update(Tree.root,1,Tree.n,++len,(t2%p+t%p)%p);
            else printf("%lld\n",t=Tree.query(Tree.root,1,Tree.n,len-t2+1,len)%p);
        }
        return;
    }
}

{

P1438 无聊的数列

link。

题面

维护一个数列，支持两种操作：

  • 1 l r K D：给出一个长度等于 $r-l+1$ 的等差数列，首项为 $K$，公差为 $D$，并将它对应加到 $[l,r]$ 范围中的每一个数上。即：令 $a_l=a_l+K,a_{l+1}=a_{l+1}+K+D,\ldots ,a_r=a_r+K+(r-l) \times D$。
  • 2 p：询问序列的第 $p$ 个数的值 $a_p$。

对于 $100%$ 数据，$0\le n,m \le 10^5,-200\le a_i,K,D\le 200, 1 \leq l \leq r \leq n, 1 \leq p \leq n$。

题解

首先我们容易想到用差分去解这道题。

我们先读入给定的等差数列，然后再开一个数组记录差分。每次 $1$ 号操作有 $L,R,D,K$ 四个参数，而我们需要进行的操作其实很简单：

  1. 区间 $[L,L]$：f[L]=f[L]+K。

  2. 区间 $(L,R]$：f[i]=f[i]+D（$i∈(L,R]$）。

  3. 区间 $[R+1,R+1]$：f[R+1]=f[R+1]-(K+((R-L)*D)。

而对于每次查询 $P$ 的值,只要输出 $\sum_{i=1}^n f_i$ 的值即可。

区间修改、区间求和。

点击查看代码 
#include<bits/stdc++.h>
using namespace std;
namespace _yz
{
    const int N=500005;
    SGT Tree;//模板 #1
    long long n,m,a[N],b[N]; 
    void yzmain()
    {
        int t1,t2,t3,t4,t5;
        scanf("%lld%lld",&n,&m);
        for(int i=1;i<=n;i++) scanf("%lld",a+i);
        for(int i=1;i<=n;i++) b[i]=a[i]-a[i-1];
        Tree.init(n,b);
        for(int i=1;i<=m;i++)
        {
            scanf("%d",&t1);
            if(t1==1)
            {
                scanf("%d%d%d%d",&t2,&t3,&t4,&t5);
                Tree.update(Tree.root,1,Tree.n,t2,t2,t4);
                if(t2!=t3) Tree.update(Tree.root,1,Tree.n,t2+1,t3,t5);
                if(t3<n) Tree.update(Tree.root,1,Tree.n,t3+1,t3+1,-(t4+t5*(t3-t2)));
            }
            else
            {
                scanf("%d",&t2);
                printf("%lld\n",Tree.query(Tree.root,1,Tree.n,1,t2));
            }
        }
        return;
    }
}
int main()
{

    //freopen("out.txt","w",stdout);
    _yz::yzmain();
    return 0;
