#include<bits/stdc++.h>
using namespace std;
struct edge
{
    int nxt,end;
}e[100000*2+5];
int cnt=0,head[100000+5];
int p1[100000+5],p2[100000+5],n,m,T;
queue<int>q;
inline void add(int a,int b)
{
    e[++cnt]=(edge){head[a],b};
    head[a]=cnt;
}
void bfs(int a)
{
    q.push(a);p1[a]=0;
    while(!q.empty())
    {
        int now=q.front();
        q.pop();
        for(int i=head[now];i;i=e[i].nxt)
        {
            int to=e[i].end;
            if(p1[to]==-1&&p2[to]==-1)
            {
                p1[to]=p1[now]+1;
                q.push(to);
            }
            else if(p1[to]!=-1&&p2[to]==-1)
            {
                if(p1[now]%2==p1[to]%2&&p1[now]!=-1)
                    p2[to]=p1[now]+1;
                else if(p2[now]%2==p1[to]%2&&p2[now]!=-1)
                    p2[to]=p2[now]+1;
                else
                    continue;
                q.push(to);
            }
            else
            {
                continue;
            }
        }
    }
    return;
}
int main()
{
    int a,b;
    scanf("%d%d%d",&n,&m,&T);
    memset(p1,-1,sizeof(p1));
    memset(p2,-1,sizeof(p2));
    for(int i=1;i<=m;i++)
        scanf("%d%d",&a,&b),add(a,b),add(b,a);
    bfs(1);
    for(int i=1;i<=T;i++)
    {
        scanf("%d%d",&a,&b);
        if(p1[a]%2==b%2&&p1[a]!=-1&&p1[a]<=b)
            printf("Yes\n");
        else if(p2[a]%2==b%2&&p2[a]!=-1&&p2[a]<=b)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}

жҺЁзҗҶ

жҲ‘д»¬еҸҜд»ҘжғіеҲ°пјҡ

  • жҲ‘д»¬иҰҒеҸ–жҜҸдёӘиҠӮзӮ№еҲ° 1 еҸ·иҠӮзӮ№еҘҮ/еҒ¶й•ҝеәҰи·Ҝеҫ„жңҖзҹӯзҡ„еҖјпјҲ$p_1$пјҢ$p_2$пјүгҖӮ
  • еҰӮжһңпјҢдёҖдёӘиҠӮзӮ№зҡ„з”ҹдә§зӯүзә§пјҲ$l$пјүеӨ§дәҺзӯүдәҺ$p_i$пјҲжҢҮ$p_1$пјҢ$p_2$дёӯзҡ„д»»ж„ҸдёҖдёӘпјҢж»Ўи¶ідёҖдёӘе°ұиЎҢпјүпјҢдё”зӯүзә§дёҺ$p_i$еҘҮеҒ¶жҖ§зӣёеҗҢпјҢеҲҷ 1 йңҖиҰҒжҸҗдҫӣеҺҹжқҗж–ҷпјҡ
    • еҪ“зӣёеҗҢж—¶пјҢжҜӢеәёзҪ®з–‘пјҢ1 дёҖе®ҡжҸҗдҫӣеҺҹжқҗж–ҷгҖӮ
    • еҪ“еӨ§дәҺ$p_i$ж—¶пјҢиӢҘдёҺ$p_i$еҘҮеҒ¶жҖ§зӣёеҗҢпјҢеҲҷеҸҜд»Ҙе®һзҺ° 1 дёҺж—Ғиҫ№зҡ„д»»ж„ҸдёҖдёӘиҠӮзӮ№иҪ®жҚўз”ҹдә§пјҢжңҖз»ҲиҪ®еҲ° 1 з”ҹдә§еҺҹжқҗж–ҷгҖӮ

иҝӣдёҖжӯҘжҺЁзҗҶпјҢжңүд»ҘдёӢдёүз§Қжғ…еҶөпјҡ

  • $l < \min(p_1,p_2)$пјҡиҪ®дёҚеҲ° 1 з”ҹдә§пјҢNoгҖӮ
  • $\min(p_1,p_2)\le l < \max(p_1,p_2)$пјҡеҰӮжһң $l = \min(a_1,a_2) \pmod 2$пјҢYesпјҢеҗҰеҲҷNoгҖӮ
  • $l \ge \max(p_1,p_2)$пјҡеҸҜд»Ҙе®һзҺ° 1 дёҺж—Ғиҫ№зҡ„д»»ж„ҸдёҖдёӘиҠӮзӮ№иҪ®жҚўз”ҹдә§пјҢжңҖз»ҲиҪ®еҲ° 1 з”ҹдә§еҺҹжқҗж–ҷгҖӮYesгҖӮ

зҺ°еңЁдјјд№ҺжӢ“еұ•е®ҢдәҶгҖӮдҪҶжҳҜпјҢжҲ‘д»¬д»…д»…жұӮеҮәдәҶp1[]ж•°з»„гҖӮ

  • еҰӮжһңp2[u]==p1[v]пјҲиҝҷж ·дҝқиҜҒp1[]p2[]еҘҮеҒ¶жҖ§дёҚеҗҢпјүпјҢиҝҷж ·p2[v]=p2[u]+1гҖӮз„¶еҗҺе°Ҷ v еҺӢе…ҘйҳҹеҲ—гҖӮ
